- success response standards

    // LIST
    { message: 'success.retrieved', data: T[], pagination?: {...} }

    // GET single
    { message: 'success.retrieved', data: T }  // T includes id, petId etc — no top-level duplication

    // CREATE
    { message: 'success.created', data: T }   // T has _id — no separate top-level id field

    // UPDATE
    { message: 'success.updated', data: T }   // or just { message } if no return body needed

    // DELETE
    { message: 'success.deleted' }
